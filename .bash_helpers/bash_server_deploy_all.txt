# Example of sser_deploy_all bash helper.

alias sser_deploy_all="
echo -e 'DEPLOY DEMO1' &&
cd ~/public_html/simpleschoolreports.se/demo1/ &&
sser_deploy_local &&
echo -e 'DEPLOY DEMO1 DONE' &&
cd ~/ &&
echo -e 'DEPLOY DEMO2' &&
cd ~/public_html/simpleschoolreports.se/demo2/ &&
sser_deploy_local &&
echo -e 'DEPLOY DEMO2 DONE' &&
cd ~/ &&
echo -e 'FULL DEPLOY DONE'
"
